## Usage

<CopyCode>
  {`
import {
  List,
  Bookmark,
  DollarSign,
  Shield,
  MessageSquare,
  Briefcase,
  Plus,
  Settings,
} from "lucide-react";
import { DropdownMenu } from "@/components/ui/DropdownMenu";

export default function DropdownExample() {
  const menuItems = [
    { icon: List, label: "Lists", id: "lists" },
    { icon: Bookmark, label: "Bookmarks", id: "bookmarks" },
    { icon: DollarSign, label: "Monetization", id: "monetization" },
    { icon: Shield, label: "Verified Orgs", id: "verified" },
    { icon: MessageSquare, label: "Ads", id: "ads" },
    { icon: Briefcase, label: "Jobs", id: "jobs" },
    { icon: Plus, label: "Create your Space", id: "create-space" },
    { icon: Settings, label: "Settings & Privacy", id: "settings" },
  ];

  return (
    <div className="min-h-screen bg-black text-white flex items-center justify-center px-4">
      <div className="bg-neutral-900 rounded-2xl shadow-lg p-6 w-full max-w-md border border-neutral-700">
        <h3 className="text-lg font-semibold mb-4 text-purple-400">Menu Example</h3>
        <div className="flex justify-center">
          <DropdownMenu
            triggerText="Open Menu"
            menuItems={menuItems}
            buttonText="Continue"
            buttonHref="/next-step"
            position="left" // or "right"
            width="w-64"
            triggerClassName="bg-purple-600 hover:bg-purple-700 text-white font-medium py-2 px-4 rounded"
            itemClassName="hover:bg-purple-800 px-4 py-3 text-sm"
          />
        </div>
      </div>
    </div>
  );
}
`}
</CopyCode>

      external: true

/>

// Disabled items
<DropdownMenu 
  menuItems={[
    { icon: User, label: "Profile", id: "profile" },
    { icon: Settings, label: "Settings", id: "settings", disabled: true }
  ]}
/>

// No action button
<DropdownMenu 
  menuItems={menuItems}
  buttonText=""
/>

// Custom styling
<DropdownMenu 
  className="custom-dropdown"
  triggerClassName="bg-blue-500 text-white hover:bg-blue-600"
  menuClassName="shadow-xl"
  itemClassName="hover:bg-blue-50"
  buttonClassName="bg-green-500 hover:bg-green-600"
  menuItems={menuItems}
/>`}
</CopyCode>

## Features

- Support for internal and external links
- Flexible positioning: left, right, center
- Disabled state support for items and trigger
- TypeScript support with comprehensive types

## MenuItem Interface

<CopyCode>
  {`interface MenuItem {
    icon: LucideIcon;     // Lucide React icon component
    label: string;        // Display text for the menu item
    id: string;          // Unique identifier
    href?: string;       // Optional URL to navigate to
    external?: boolean;  // Whether to open link in new tab
    disabled?: boolean;  // Whether the item is disabled
}`}
</CopyCode>
