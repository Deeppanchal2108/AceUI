## Usage

<CopyCode>
  {`import { DropdownMenu } from "@/components/dropdown-menu";
import { Settings, User, LogOut, Plus } from "lucide-react";

const menuItems = [
  { icon: User, label: "Profile", id: "profile", href: "/profile" },
  { icon: Settings, label: "Settings", id: "settings", href: "/settings" },
  { icon: LogOut, label: "Sign Out", id: "signout", href: "/logout" }
];

export default function Example() {
  return (
    <div>
      <DropdownMenu 
        triggerIcon={Plus}
        triggerText="Actions"
        menuItems={menuItems}
        buttonText="Create Post"
        buttonHref="/create"
      />
    </div>
  );
}`}
</CopyCode>

  menuItems={menuItems}
/>

<DropdownMenu 
  menuItems={[
    { 
      icon: ExternalLink, 
      label: "Documentation", 
      id: "docs", 
      href: "https://docs.example.com",
      external: true 
    }
  ]}
/>

// Disabled items
<DropdownMenu 
  menuItems={[
    { icon: User, label: "Profile", id: "profile" },
    { icon: Settings, label: "Settings", id: "settings", disabled: true }
  ]}
/>

// No action button
<DropdownMenu 
  menuItems={menuItems}
  buttonText=""
/>

// Custom styling
<DropdownMenu 
  className="custom-dropdown"
  triggerClassName="bg-blue-500 text-white hover:bg-blue-600"
  menuClassName="shadow-xl"
  itemClassName="hover:bg-blue-50"
  buttonClassName="bg-green-500 hover:bg-green-600"
  menuItems={menuItems}
/>`}
</CopyCode>

## Notice

The component uses `window.location.href` for internal navigation. For Next.js applications, consider extending it to use `next/router` or `next/navigation` for better performance and user experience.

## MenuItem Interface

<CopyCode>
  {`interface MenuItem {
    icon: LucideIcon;     // Lucide React icon component
    label: string;        // Display text for the menu item
    id: string;          // Unique identifier
    href?: string;       // Optional URL to navigate to
    external?: boolean;  // Whether to open link in new tab
    disabled?: boolean;  // Whether the item is disabled
}`}
</CopyCode>
