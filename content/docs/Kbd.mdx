## Variants

<CopyCode>
  {`// Default variant
<Kbd keys={["ctrl", "c"]} />

// Outline variant  
<Kbd keys={["shift", "enter"]} variant="outline" />

// Ghost variant
<Kbd keys={["escape"]} variant="ghost" />

// Different sizes
<Kbd keys={["tab"]} size="sm" />
<Kbd keys={["space"]} size="md" />
<Kbd keys={["command", "k"]} size="lg" />

// Single keys
<Kbd keys={["enter"]} />
<Kbd keys={["delete"]} />
<Kbd keys={["escape"]} />

// Multiple key combinations
<Kbd keys={["command", "shift", "p"]} />
<Kbd keys={["ctrl", "alt", "delete"]} />

// Arrow keys
<Kbd keys={["up"]} />
<Kbd keys={["down", "right"]} />

// Custom text keys
<Kbd keys={["F1"]} />
<Kbd keys={["a", "b", "c"]} />`}
</CopyCode>
